function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet()
              .getSheetByName('TMLF_Training') 
              || SpreadsheetApp.getActiveSpreadsheet().insertSheet('TMLF_Training');
  var data = JSON.parse(e.postData.contents);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      'timestamp','sessionName','trainerName','meetLink',
      'participantName','teamName','activityType','questionText',
      'responseStatus','pointsAwarded','participationStatus','notes'
    ]);
  }
  data.rows.forEach(function(row) {
    sheet.appendRow([
      row.timestamp, row.sessionName, row.trainerName, row.meetLink,
      row.participantName, row.teamName, row.activityType, row.questionText,
      row.responseStatus, row.pointsAwarded, row.participationStatus, row.notes
    ]);
  });
  return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
         .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  return ContentService.createTextOutput('TMLF Training Command Center — Online')
         .setMimeType(ContentService.MimeType.TEXT);
}
